```python
# custom_filter.py
from rsj_ffmpeg.plugin import Plugin

class CustomFilter(Plugin):
    name = "rsj_custom_filter"
    version = "1.0.0"
    
    def process(self, input_file, output_file, **kwargs):
        # Your custom processing logic
        self.apply_filter(input_file, "custom_effect")
        self.add_watermark("RAJSARASWATI JATAV")
        self.save(output_file)
        
    def get_ffmpeg_args(self):
        return ["-vf", "custom_filter_chain"]

# Register plugin
```

```javascript
// custom_overlay.js
const { Plugin } = require('rsj-ffmpeg');

class CustomOverlay extends Plugin {
  constructor() {
    super();

    this.version = '1.0.0';
  }
  
  process(inputFile, outputFile, options) {
    // Your custom overlay logic

    this.save(outputFile);
  }
}

module.exports = CustomOverlay;
```
