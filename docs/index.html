Usage Examples

See it in action

# Convert video
rsj-ffmpeg -i input.mp4 -o output.mp4

# AI upscale with watermark
rsj-ffmpeg -i video.mp4 --ai-upscale 4x --watermark "RSJ" -o out.mp4

# Batch processing
rsj-ffmpeg --batch ./videos/ --enhance --export ./output/

# Create GIF
rsj-ffmpeg -i video.mp4 --to-gif -o animation.gif
from rsj_ffmpeg import RSJToolkit

toolkit = RSJToolkit()

# Batch convert
toolkit.batch_convert(
    input_dir="./videos/",
    output_dir="./output/",
    ai_upscale="4x",
    enhance=True,
    watermark="RAJSARASWATI JATAV"
)

# AI enhancement
toolkit.ai_enhance(
    input_file="video.mp4",
    output_file="enhanced.mp4",
    upscale=4
)
# Start API server
python3 api_server.py

# Convert video
curl -X POST http://localhost:8080/api/v1/convert \
  -H "Content-Type: application/json" \
  -d '{"input_file": "input.mp4", "output_file": "output.mp4"}'

# Start stream
curl -X POST http://localhost:8080/api/v1/stream/start \
  -H "Content-Type: application/json" \
  -d '{"platform": "youtube", "key": "YOUR_KEY"}'
# Telegram Bot Commands
/start - Initialize bot
/convert mp4 - Convert video
/enhance - AI enhance
/watermark TEXT - Add watermark
/gif - Create GIF
/audio - Extract audio
